vector<int> topological_sort(int n) {
    queue<int> q;
    vector<int> seq;

    for (int u = 1; u <= n; u++)
        if (in[u] == 0)
            q.push(u);

    while (!q.empty()) {
        int u = q.front();
        q.pop();

        seq.push_back(u);

        for (int v : edges[u])
            if (--in[v] == 0)
                q.push(v);
    }
    return seq;
}

void DFS(int u, vector<int>& seq) {
    vis[u] = true;
    for (int v : edges[u])
        if (!vis[v])
            DFS(v, seq);
    seq.push_back(u);
}

vector<int> topological_sort() {
    vector<int> seq;

    for (int u = 1; u <= n; u++)
        if (!vis[u])
            DFS(u, seq);

    reverse(seq.begin(), seq.end());
    return seq;
}
